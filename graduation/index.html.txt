<script>
    const canvas = document.getElementById('canvas');
    const ctx = canvas.getContext('2d');
    const baseImage = new Image();
    const userImage = new Image();
    let userImageLoaded = false;

    // 1. الصورة الأساسية (خلفية التخرج)
    baseImage.src = 'grad-card-bg.jpg'; 

    // تحديث نص الرفع عند اختيار صورة
    document.getElementById('imageUpload').onchange = function(e) {
        const file = e.target.files[0];
        if (file) {
            const reader = new FileReader();
            reader.onload = function(event) {
                userImage.src = event.target.result;
                userImage.onload = () => { userImageLoaded = true; };
                document.getElementById('uploadStatus').innerText = "✅ تم اختيار الصورة";
                document.getElementById('uploadStatus').classList.add('text-emerald-600');
            };
            reader.readAsDataURL(file);
        }
    };

    async function generateGraduationCard() {
        const name = document.getElementById('nameInput').value.trim();
        
        if (!name) {
            Swal.fire({ icon: 'error', title: 'عفواً', text: 'يرجى كتابة اسم الخريج', confirmButtonColor: '#0c4a6e' });
            return;
        }

        // إعدادات الحجم
        canvas.width = baseImage.width || 1000; 
        canvas.height = baseImage.height || 1400;

        // رسم الخلفية
        ctx.drawImage(baseImage, 0, 0, canvas.width, canvas.height);

        // 2. إضافة صورة الخريج (إذا تم رفعها)
        if (userImageLoaded) {
            drawUserImage();
        }

        // 3. إضافة النص
        ctx.font = "bold 60px 'BukraFont'";
        ctx.fillStyle = "#0c4a6e"; // لون كحلي طبي
        ctx.textAlign = "center";
        // مكان الاسم (يعدل حسب تصميم خلفيتك)
        ctx.fillText(name, canvas.width / 2, canvas.height * 0.85); 

        // إظهار النتائج
        const wrapper = document.getElementById('canvasWrapper');
        wrapper.classList.remove('hidden');
        setTimeout(() => {
            wrapper.classList.add('scale-100', 'opacity-100');
            document.getElementById('downloadBtn').style.display = 'flex';
        }, 100);

        confetti({ particleCount: 150, spread: 100, origin: { y: 0.7 } });
    }

    function drawUserImage() {
        // إحداثيات صورة الخريج (يفترض أنها في المنتصف)
        const size = 300; // حجم الصورة الشخصية
        const x = (canvas.width / 2) - (size / 2);
        const y = 300; // الارتفاع من الأعلى

        ctx.save();
        // رسم دائرة لقص الصورة داخلها (Clip)
        ctx.beginPath();
        ctx.arc(canvas.width / 2, y + (size/2), size / 2, 0, Math.PI * 2);
        ctx.closePath();
        ctx.clip();

        // رسم الصورة الشخصية
        ctx.drawImage(userImage, x, y, size, size);
        ctx.restore();

        // رسم إطار حول الدائرة
        ctx.strokeStyle = "#0c4a6e";
        ctx.lineWidth = 10;
        ctx.stroke();
    }

    function downloadCard() {
        const link = document.createElement('a');
        link.download = 'تخرج_نادي_التمريض.png';
        link.href = canvas.toDataURL('image/png');
        link.click();
    }
</script>
